#[macro_use]
extern crate rocket;

#[get("/")]
fn index() -> &'static str {
    "Hello, world!"
}

#[get("/out_data")] // <- rquest attribute
fn out_data() -> &'static str {
    // request handeler /\
    "Test"
}

#[launch]
fn rocket() -> _ {
    rocket::build()
        .mount("/", routes![index])
        .mount("/out_data", routes![out_data])
        .launch() // Call launch() to finalize the build and return Rocket<Build>
}
